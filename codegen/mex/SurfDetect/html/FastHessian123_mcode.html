1function responseMap = FastHessian(intImage)  %#codegen
 2
 3%   Copyright 2017 The MathWorks, Inc.
 4%
 5% This function constructs the scale-space by convolving the integral
 6% image with box filter approximations of various sizes. The following
 7% constants are used:
 8%
 9% Number of Octaves: 5
10% Number of Intervals: 4
11% Initial Step Size: 2
12% Threshold: 0.0004
13% Filter Sizes: Octave 1 -  9,  15,  21,  27
14%               Octave 2 - 15,  27,  39,  51
15%               Octave 3 - 27,  51,  75,  99
16%               Octave 4 - 51,  99, 147, 195
17%               Octave 5 - 99, 195, 291, 387
18
19% Image dimensions
20imgDim   = size(intImage);
21i_height = imgDim(1);
22i_width  = imgDim(2);
23
24% Initialize response map for various octaves and intervals
25init_sample = 2;
26w           = floor(i_width/init_sample);
27h           = floor(i_height/init_sample);
28s           = init_sample;
29
30responseMap =  coder.nullcopy([
31    struct('width',w          ,'height',h          ,'step',s   ,'filter',9  ,'responses',zeros(h,w                    ,'single'),'laplacian',zeros(h,w                    ,'uint8')) ...
32    struct('width',w          ,'height',h          ,'step',s   ,'filter',15 ,'responses',zeros(h,w                    ,'single'),'laplacian',zeros(h,w                    ,'uint8')) ...
33    struct('width',w          ,'height',h          ,'step',s   ,'filter',21 ,'responses',zeros(h,w                    ,'single'),'laplacian',zeros(h,w                    ,'uint8')) ...
34    struct('width',w          ,'height',h          ,'step',s   ,'filter',27 ,'responses',zeros(h,w                    ,'single'),'laplacian',zeros(h,w                    ,'uint8')) ...
35    struct('width',floor(w/2) ,'height',floor(h/2) ,'step',s*2 ,'filter',39 ,'responses',zeros(floor(h/2),floor(w/2)  ,'single'),'laplacian',zeros(floor(h/2),floor(w/2)  ,'uint8')) ...
36    struct('width',floor(w/2) ,'height',floor(h/2) ,'step',s*2 ,'filter',51 ,'responses',zeros(floor(h/2),floor(w/2)  ,'single'),'laplacian',zeros(floor(h/2),floor(w/2)  ,'uint8')) ...
37    struct('width',floor(w/4) ,'height',floor(h/4) ,'step',s*4 ,'filter',75 ,'responses',zeros(floor(h/4),floor(w/4)  ,'single'),'laplacian',zeros(floor(h/4),floor(w/4)  ,'uint8')) ...
38    struct('width',floor(w/4) ,'height',floor(h/4) ,'step',s*4 ,'filter',99 ,'responses',zeros(floor(h/4),floor(w/4)  ,'single'),'laplacian',zeros(floor(h/4),floor(w/4)  ,'uint8')) ...
39    struct('width',floor(w/8) ,'height',floor(h/8) ,'step',s*8 ,'filter',147,'responses',zeros(floor(h/8),floor(w/8)  ,'single'),'laplacian',zeros(floor(h/8),floor(w/8)  ,'uint8')) ...
40    struct('width',floor(w/8) ,'height',floor(h/8) ,'step',s*8 ,'filter',195,'responses',zeros(floor(h/8),floor(w/8)  ,'single'),'laplacian',zeros(floor(h/8),floor(w/8)  ,'uint8')) ...
41    struct('width',floor(w/16),'height',floor(h/16),'step',s*16,'filter',291,'responses',zeros(floor(h/16),floor(w/16),'single'),'laplacian',zeros(floor(h/16),floor(w/16),'uint8')) ...
42    struct('width',floor(w/16),'height',floor(h/16),'step',s*16,'filter',387,'responses',zeros(floor(h/16),floor(w/16),'single'),'laplacian',zeros(floor(h/16),floor(w/16),'uint8')) ]);
43
44responseMap(1).width=w;           responseMap(1).height=h         ;responseMap(1).step=s   ;responseMap(1).filter=9;
45responseMap(2).width=w          ;responseMap(2).height=h          ;responseMap(2).step=s   ;responseMap(2).filter=15;
46responseMap(3).width=w          ;responseMap(3).height=h          ;responseMap(3).step=s   ;responseMap(3).filter=21;
47responseMap(4).width=w          ;responseMap(4).height=h          ;responseMap(4).step=s   ;responseMap(4).filter=27;
48responseMap(5).width=floor(w/2) ;responseMap(5).height=floor(h/2) ;responseMap(5).step=s*2 ;responseMap(5).filter=39;
49responseMap(6).width=floor(w/2) ;responseMap(6).height=floor(h/2) ;responseMap(6).step=s*2 ;responseMap(6).filter=51;
50responseMap(7).width=floor(w/4) ;responseMap(7).height=floor(h/4) ;responseMap(7).step=s*4 ;responseMap(7).filter=75;
51responseMap(8).width=floor(w/4) ;responseMap(8).height=floor(h/4) ;responseMap(8).step=s*4 ;responseMap(8).filter=99;
52responseMap(9).width=floor(w/8) ;responseMap(9).height=floor(h/8) ;responseMap(9).step=s*8 ;responseMap(9).filter=147;
53responseMap(10).width=floor(w/8) ;responseMap(10).height=floor(h/8) ;responseMap(10).step=s*8 ;responseMap(10).filter=195;
54responseMap(11).width=floor(w/16);responseMap(11).height=floor(h/16);responseMap(11).step=s*16;responseMap(11).filter=291;
55responseMap(12).width=floor(w/16);responseMap(12).height=floor(h/16);responseMap(12).step=s*16;responseMap(12).filter=387;
56
57heightArray = [h floor(h/2) floor(h/4) floor(h/8) floor(h/16)];
58 widthArray = [w floor(w/2) floor(w/4) floor(w/8) floor(w/16)];
59
60% Compute box filter convolution for 12 unique filter sizes
61for i = coder.unroll(1:12)
62    
63    switch i
64        case {1,2,3,4}
65            index = 1;
66        case {5,6}
67            index = 2;
68        case {7,8}
69            index = 3;
70        case {9,10}
71            index = 4;
72        otherwise
73            index = 5;
74    end
75    
76    height = heightArray(index);
77    width  = widthArray(index);
78    filter = responseMap(i).filter;
79    step   = responseMap(i).step;
80    
81    [respMatrix, lapMatrix] = FastHessianCalc(intImage, height, width, step, filter);
82    
83    responseMap(i).responses = respMatrix;
84    responseMap(i).laplacian = lapMatrix;
85    
86end
87
88end
