1function [respMatrix, lapMatrix] = FastHessianCalc(intImage, height, width, step, filter) %#codegen
 2
 3%   Copyright 2017 The MathWorks, Inc.
 4
 5    respMatrix = zeros(height, width,'single');
 6    lapMatrix  = zeros(height, width,'uint8');
 7
 8    bor = (filter - 1) / 2;     % border size for this filter
 9    l   = filter / 3;           % lobe size for this filter 
10    w   = filter;               % filter size
11    inverse_area = 1/(w*w);     % normalization factor
12    
13    coder.gpu.kernel();
14    for ac = 0:width-1
15        
16        for ar = 0:height-1
17        
18            % get the image coordinates
19            r = ar * step;
20            c = ac * step;
21            
22            % Compute elements of the Hessian Matrix
23            Dxx = BoxIntegral(intImage, r - l + 1, c - bor, 2*l - 1, w) ...
24                - BoxIntegral(intImage, r - l + 1, ceil(c - l / 2), 2*l - 1, l)*3;
25            
26            Dyy = BoxIntegral(intImage, r - bor, c - l + 1, w, 2*l - 1) ...
27                - BoxIntegral(intImage, ceil(r - l / 2), c - l + 1, l, 2*l - 1)*3;
28            
29            Dxy = BoxIntegral(intImage, r - l, c + 1, l, l) ...
30                + BoxIntegral(intImage, r + 1, c - l, l, l) ...
31                - BoxIntegral(intImage, r - l, c - l, l, l) ...
32                - BoxIntegral(intImage, r + 1, c + 1, l, l);
33            
34            % Normalize the filter responses with respect to their size
35            Dxx = Dxx * inverse_area;
36            Dyy = Dyy * inverse_area;
37            Dxy = Dxy * inverse_area;
38            
39            % Get the determinant of hessian response & laplacian sign
40            respMatrix(ar+1,ac+1) = (Dxx * Dyy - single(0.81) * Dxy * Dxy);
41            
42            if Dxx + Dyy >= 0
43                lapMatrix(ar+1,ac+1) = uint8(1);
44            else
45                lapMatrix(ar+1,ac+1) = uint8(0);
46            end
47            
48        end
49        
50    end
51    
52end
